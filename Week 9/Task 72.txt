class Solution {
    public int reverseBits(int n) {

        int result = 0;

        for (int i = 0; i < 32; i++) {

            int bit = n & 1;        // extract last bit
            result = (result << 1) | bit;  // add to result
            n = n >> 1;             // shift input

        }

        return result;
    }
}
